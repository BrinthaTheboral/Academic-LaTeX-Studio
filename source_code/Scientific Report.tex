	\section{Mathematical Model}
	
	Air quality prediction function:
	
	\[
	AQI_t = f(X_t,W,b)
	\]
	
	Where:
	
	\begin{itemize}
		
		\item $AQI_t$ = Air Quality Index
		
		\item $X_t$ = Environmental sensor data
		
		\item $W$ = Weight parameters
		
		\item $b$ = Bias
		
	\end{itemize}
	
	Mean squared error:
	
	\[
	MSE = \frac{1}{n}\sum_{i=1}^{n}(y_i - \hat{y_i})^2
	\]
