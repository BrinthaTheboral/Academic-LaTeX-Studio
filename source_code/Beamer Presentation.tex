	% =================================================
	
	\begin{frame}{Mathematical Model}
		
		Traffic prediction function:
		
		\[
		Y_t = f(X_t,W,b)
		\]
		
		\vspace{0.5cm}
		
		Loss function:
		
		\[
		L = \frac{1}{n}\sum_{i=1}^{n}(y_i - \hat{y_i})^2
		\]
		
		\vspace{0.5cm}
		
		Where:
		
		\begin{itemize}
			
			\item $X_t$ = Input traffic features
			
			\item $W$ = Weight parameters
			
			\item $b$ = Bias term
			
		\end{itemize}
		
	\end{frame}
